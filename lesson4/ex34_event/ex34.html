var majorBtn = document.getElementById("major");
            var bind = document.getElementById("bind");
            var unbind = document.getElementById("unbind");
            bind.onclick = function () {
                majorBtn.addEventListener("click", clickEvent, false);
                majorBtn.innerHTML = "useful button";
            };
            unbind.onclick = function () {
                majorBtn.removeEventListener("click", clickEvent, false);
                majorBtn.innerHTML = "useless button";
            };


            function clickEvent() {
                alert("clicked")
            }
